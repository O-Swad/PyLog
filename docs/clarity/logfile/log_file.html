import os
import time
from collections import Counter
from clarity.logline.log_line import LogLine


class LogFile(object):
    """
    Class for managing log files
    """

    def __init__(self, name):
        self._name = name
        """Name of the file"""
        self._lines = list()
        """Lines composed by timestamp host_from and host_to"""

    @property
    def name(self):
        return self._name

    @name.setter
    def name(self, value):
        self._name = value

    @property
    def lines(self):
        return self._lines

    @lines.setter
    def lines(self, value):
        self._lines = value

    def parse(self, buffer_size=512):
        """
        implements a reading algorithm based on chunks from the end of file. if the file size is less than 1Mb it is
        dumped to ram, else a buffer is used
        """
        # file size
        file_size = os.stat(self._name).st_size
        # a chunk = block_number * buffer size
        block_number = 0
        last_lines_for_tokenize = None
        data = []

        try:
            with open(self._name) as file:

                # buffer size can't be greater than file size for not dumping all the file in RAM
                if buffer_size > file_size:
                    buffer_size = file_size - 1

                # file size greater than 1Mb then use a buffer for reading in chunks
                if file_size > 1000000:
                    while True:
                        data = []
                        block_number += 1

                        # go directly to the file position determined by buffer_size and block_number
                        file.seek(file_size - buffer_size * block_number)
                        # load in RAM all lines from seek position to the end of the file
                        data.extend(file.readlines())

                        # when no more for reading, exit
                        if file.tell() == 0:
                            last_lines_for_tokenize = data
                            break
                # tiny file, then dump into ram
                else:
                    data.extend(file.readlines())
                    last_lines_for_tokenize = data
        except IOError as error:
            print(error)

        self._to_tokenized_lines(last_lines_for_tokenize)

    def _to_tokenized_lines(self, data):
        """
        :param data: data list from file for tokenizing
        :return: a list of LogLines
        """
        self._lines.clear()
        if data:

            for line in data:
                t_stamp = line.split()[0]
                host_ori = line.split()[1]
                host_dest = line.split()[2]
                self._lines.append(LogLine(t_stamp, host_ori, host_dest))

    def _get_last_hour(self) -> list:
        """
        get the log lines of the last hour
        :return: A list of tuples with the format (from_host, to_host)
        """

        # current time in milliseconds
        now_time = int(time.time()*1000)

        # filter the lines from the last hour
        hour_filter = filter(lambda x: int(x.log_timestamp) >= now_time - 3600000, self._lines)

        # to create list of tuples [(from, to),...]
        loglines_list = []
        for line in hour_filter:
            loglines_list.append((line.host_ori, line.host_dest))

        return loglines_list

    def _get_between(self, init_datetime: int, end_datetime: int) -> list:
        """
        get the log lines in a period
        :param init_datetime start time of the period
        :param end_datetime end time of the period
        """

        period_filter = list(filter(
                                    lambda x:
                                    init_datetime <= int(x.log_timestamp) <= end_datetime, self._lines
                                    )
                             )

        loglines_list = []
        for line in period_filter:
            loglines_list.append((line.host_ori, line.host_dest))

        return loglines_list

    def get_host_connected_to_in_period(self, init_datetime: int, end_datetime: int, host_to: str) -> list:
        """
        get a list of hostnames connected to the given host during the given period
        :param init_datetime start time of the period
        :param end_datetime end time of the period
        :param host_to given host
        """
        to_filter = list(filter(lambda x: x[1] == host_to, self._get_between(init_datetime, end_datetime)))
        hosts_connected_to = (host for host in Counter([x[0] for x in to_filter]).most_common())
        return hosts_connected_to

    def get_hosts_connected_to(self, host_to: str) -> list:
        """
        get a list of hostnames connected to a given host
        :param host_to: given host
        :param hosts: list of hosts in the last hour
        :return: list of hostnames connected to the given host
        """
        to_filter = list(filter(lambda x: x[1] == host_to, self._get_last_hour()))
        hosts_connected_to = (host for host in Counter([x[0] for x in to_filter]).most_common())
        return hosts_connected_to

    def get_hosts_connected_from(self, host_from: str) -> list:
        """
        get a list of hostnames connected from a given host
        :param host_from: given host
        :return: list of hostnames connected from the given host
        """

        from_filter = filter(lambda x: x[0] == host_from, self._get_last_hour())
        hosts_connected_from = (host for host in Counter([x[1] for x in from_filter]).most_common())
        return hosts_connected_from

    def get_most_connections_host(self) -> str:
        """
        get the host which opened more connections (from hosts)
        :param hosts: list of hosts in the last hour
        :return: list of hostnames connected to the given host
        """
        return [host for host in Counter([x[0] for x in self._get_last_hour()]).most_common(1)]
